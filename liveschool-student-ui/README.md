# 安装依赖
yarn install
# 启动
yarn run serve     
# 编译
npm run build
#项目结构
```
|-- liveschool-student-ui
    |-- package-lock.json
    |-- package.json //模块的描述文件
    |-- vue.config.js //vue配置文件
    |-- dist //编译后的目录文件夹
    |-- public //静态资源路径
    |   |-- index.html
    |-- src //
        |-- main.js
        |-- config
            |-- env.js //系统接口请求配置
```
